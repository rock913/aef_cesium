一、 算法调优逻辑：从“过拟合”到“弹性泛化”

在前期 V8.0 版本的强共识交叉验证中，我们保障了样本的“绝对纯净”，但忽略了真实地理环境的“灰度渐变区”，导致模型面临局部过拟合与碎片化噪点的风险。

V8.1 版本引入了弹性控制机制 (Elastic Control)，旨在平衡“局部像素精度”与“宏观地貌泛化性”：

样本泛化 (Soft Margin)：释放过于严苛的掩膜条件，容许特征空间存在合理的“软边界”，覆盖真实海岸带的过渡态特征。

树的正则化 (Tree Regularization)：通过限制随机森林的树深与叶子节点容量，阻断算法死记硬背噪声，提升在未知海域的泛化能力。

形态学约束 (Morphological Smoothing)：在地学空间维度注入“平滑算子”，抹除非自然的椒盐碎斑，提升审计级制图的视觉连贯性和最终物理准确性。

"""Chapter 5 (Yancheng coastline audit) V8.1 Generalized Consensus Export."""
from __future__ import annotations
import argparse
import os
import time
from typing import Any, Dict
import ee

def _resolve_asset_id() -> str:
    explicit = str(os.getenv("CH5_RF_ASSET_ID", "")).strip()
    if explicit: return explicit
    gee_user_path = str(os.getenv("GEE_USER_PATH", "")).strip()
    if gee_user_path and gee_user_path != "users/default/aef_demo":
        return f"{gee_user_path.rstrip('/')}/classifiers/ch5_coastline_rf_v1"
    return ""

def _init_ee() -> None:
    service_account = str(os.getenv("EE_SERVICE_ACCOUNT", "")).strip()
    private_key_file = str(os.getenv("EE_PRIVATE_KEY_FILE", "")).strip()
    if service_account and private_key_file:
        creds = ee.ServiceAccountCredentials(service_account, private_key_file)
        ee.Initialize(creds)
    else:
        ee.Initialize()

def _ensure_asset_folders_exist(asset_id: str) -> None:
    parent = asset_id.rsplit("/", 1)[0] if "/" in asset_id else ""
    if not parent: return
    try:
        ee.data.createAsset({"type": "FOLDER"}, parent)
    except Exception:
        pass

def _build_classifier_graph() -> Any:
    # 1. 基础特征
    emb_col = ee.ImageCollection("GOOGLE/SATELLITE_EMBEDDING/V1/ANNUAL")
    emb_bands = [f"A{idx:02d}" for idx in range(0, 16)]
    base_img = emb_col.filterDate("2023-01-01", "2025-01-01").mosaic().select(list(range(0, 16))).rename(emb_bands)

    # 2. 金标准引入
    esa = ee.ImageCollection("ESA/WorldCover/v200").first().select('Map')
    jrc = ee.Image("JRC/GSW1_4/GlobalSurfaceWater").select('occurrence') 

    # 3. 🛡️ 泛化性调优：软边界共识掩膜 (Soft Margin Consensus)
    # 适度放宽绝对值，包容极端汛期/干旱导致的局部历史偏差，使得训练集覆盖更多过渡态，增强模型泛化力。
    
    # Class 0: 水体 (将 JRC > 90 放宽到 JRC >= 80，涵盖潮间带下部的高频淹没区)
    pure_water = jrc.gte(80).And(esa.eq(80))
    
    # Class 1: 滩涂 (覆盖 5%~80% 的广阔潮间带变异度)
    pure_flat = jrc.gt(5).And(jrc.lt(80)).And(esa.neq(50))
    
    # Class 2: 人工围填海与堤坝 (将 JRC == 0 放宽到 <= 2，包容 ESA 建筑区内可能存在极小水坑的情况)
    pure_built = esa.eq(50).And(jrc.unmask(0).lte(2))
    
    # Class 3: 绝对内陆
    pure_inland = esa.eq(40).Or(esa.eq(10)).And(jrc.unmask(0).lte(2))

    labels = ee.Image(-1) \
        .where(pure_water, 0) \
        .where(pure_flat, 1) \
        .where(pure_built, 2) \
        .where(pure_inland, 3)

    training_mask = labels.neq(-1)
    gold_labels = labels.updateMask(training_mask).rename('class')

    training_stack = base_img.addBands(gold_labels)
    training_region = ee.Geometry.Rectangle([119.8, 32.8, 121.5, 34.0])
    
    scale = int(os.getenv("CH5_RF_SAMPLE_SCALE", "30") or "30")
    # 增加样本量以吸收更多方差
    training_data = training_stack.stratifiedSample(
        numPoints=3000, 
        classBand='class',
        region=training_region,
        scale=scale,
        seed=42,
        tileScale=4,
        geometries=False
    )

    # 4. 🛡️ 精度与泛化控制：Random Forest 正则化 (Regularization)
    # - numberOfTrees=60: 增加树的数量，利用群体决策提升整体稳定性
    # - minLeafPopulation=10: 核心泛化参数！禁止树分枝到纯度极高的单一像素，强迫模型学习宏观模式而非死记硬背局部噪点
    # - bagFraction=0.6: 每棵树只使用 60% 的数据，增加基学习器的差异性，有效抵抗过拟合
    classifier = ee.Classifier.smileRandomForest(
        numberOfTrees=60, 
        minLeafPopulation=10, 
        bagFraction=0.6 
    ).train(
        features=training_data, classProperty="class", inputProperties=emb_bands,
    )
    return classifier

def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--asset-id", default="", help="Override target asset id")
    parser.add_argument("--ensure", action="store_true", help="Submit export task")
    args = parser.parse_args()

    asset_id = str(args.asset_id).strip() or _resolve_asset_id()
    if not asset_id: return 2
    
    _init_ee()
    if not args.ensure:
        print(f"Asset target: {asset_id}. Use --ensure to export.")
        return 0

    _ensure_asset_folders_exist(asset_id)
    task = ee.batch.Export.classifier.toAsset(
        classifier=_build_classifier_graph(), description="Export_Coastline_RF_V8_1", assetId=asset_id
    )
    task.start()
    print(f"✅ V8.1 泛化调优任务已提交: {asset_id}")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
